$().ready(function(){
    $(document).keypress(function(key){
        if(key.which == 114)
            $("p.par").css("color", "red");
        else if(key.which == 98)
            $("p.par").css("color", "blue");
        else if(key.which == 103)
            $("p.par").css("color", "green");
    });
});
